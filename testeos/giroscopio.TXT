#include <Wire.h>
#include <MPU6050.h>

MPU6050 mpu;

int16_t ax, ay, az;
int16_t gx, gy, gz;
int16_t prev_gz = 0;
int16_t delta_gz = 0;
float currentAngle = 0.0;
float desiredAngle = 90.0;

int IN1 = 2; 
int IN2 = 3; 
int IN3 = 4; 
int IN4 = 5; 

bool isTurning = false;
unsigned long turnStartTime = 0;
const unsigned long turnDuration = 1000; // 1000 milisegundos = 1 segundo

void setup() {
  Wire.begin();
  Serial.begin(9600);

  mpu.initialize();

  mpu.setFullScaleGyroRange(MPU6050_GYRO_FS_250);
  mpu.setDLPFMode(MPU6050_DLPF_BW_42);

  pinMode(IN1, OUTPUT);
  pinMode(IN2, OUTPUT);
  pinMode(IN3, OUTPUT);
  pinMode(IN4, OUTPUT);
}

void loop() {
  mpu.getMotion6(&ax, &ay, &az, &gx, &gy, &gz);

  delta_gz = gz - prev_gz;
  prev_gz = gz;

  currentAngle += delta_gz * 0.0000611; // 0.0000611 = 1 / (250 Hz)

  if (!isTurning && currentAngle < desiredAngle) {
    // Comenzar el giro
    isTurning = true;
    turnStartTime = millis();
    digitalWrite(IN1, HIGH);
    digitalWrite(IN2, LOW);
    digitalWrite(IN3, LOW);
    digitalWrite(IN4, HIGH);
  } else if (isTurning && (millis() - turnStartTime) >= turnDuration || currentAngle >= desiredAngle) {
    // Detener el giro
    isTurning = false;
    digitalWrite(IN1, LOW);
    digitalWrite(IN2, LOW);
    digitalWrite(IN3, LOW);
    digitalWrite(IN4, LOW);
  }

  Serial.print("√Ångulo actual: ");
  Serial.println(currentAngle);
}
